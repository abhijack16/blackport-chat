// Start Winsock version 2.2 and fill wsaData with startup information.
if (WSAStartup(MAKEWORD(2,2), &wsaData) != 0)
{
    std::cout << "WSAStartup failed\n";

    // Exit program because networking cannot work without Winsock.
    return 1;
}


// Create a TCP socket using IPv4.

// socket() returns a socket handle which is stored in clientSocket.

clientSocket = socket(AF_INET, SOCK_STREAM, 0);


// Check whether socket creation succeeded.
if(clientSocket == INVALID_SOCKET)
{
    std::cout << "Socket creation failed\n";

    // Exit because communication is impossible without a valid socket.
    return 1;
}
